\defverbatim[colored]\LstNestedGoto{%
\begin{lstlisting}
int i,j,v;
for(i=1,v=1; i<=8; i+=3){
    for(j=3; j<6; j++){
        cout<<i<<" "<<j<<endl;
        cin>>v;
        if(v==0){
            goto point;
        }
    }
}
point: //continue from here;
\end{lstlisting}
}
